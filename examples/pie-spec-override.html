Style Override

Pie Chart: Override styles in the Specification

Demonstrate how to override the specification such as changing colors or adding legend or marks.

DC Schools that are District? 
<html>
<head>
  <meta charset=utf-8 />
  <title>Style Override</title>
  <meta name='viewport' content='initial-scale=1,maximum-scale=1,user-scalable=no' />

  <!-- Load D3 and vega -->
  <script type="text/javascript" src="http://cdnjs.cloudflare.com/ajax/libs/d3/3.5.5/d3.min.js"></script>
  <script type="text/javascript" src="http://cdnjs.cloudflare.com/ajax/libs/vega/2.5.2/vega.min.js"></script>

  <!-- Load Cedar -->
  <script type="text/javascript" src="cedar.js"></script>

</head>
<body>

<div class="row">
  <div class="col-lg-12" id="chart"></div>
</div>

<script>

  //create a cedar chart, passing a url to a spec
  var chart = new Cedar({"specification":"../data/templates/pie.json"});
  //create the dataset w/ mappings
  var dataset = {
    "url":"https://services.arcgis.com/uDTUpUPbk8X8mXwl/arcgis/rest/services/Public_Schools_in_Onondaga_County/FeatureServer/0",
    "mappings":{
      "group": {"field":"District","label":"District"},
      "count": {"field":"Number_of","label":"Student Number_of"}
    }
  };

  chart.override = {
    "marks": [{"properties": {
      "hover": {"fill": {"value": "orange"}},
      "update": {"fill": {"value": "#c15000"}}}}]};


  //assign to the chart
  chart.dataset = dataset;

  //show the chart
  chart.show({
    elementId: "#chart"
  });

</script>
</body>
</html>
